function shuffle(a){ return [...a].sort(()=>Math.random()-0.5); }
function $(id){ return document.getElementById(id); }

function calcLevel(xp){
  let lv=0;
  for(let i=0;i<LEVELS.length-1;i++){
    if(xp>=LEVELS[i].max) lv=i+1; else break;
  }
  return lv;
}

function xpInLevel(xp,lv){
  const base = lv>0?LEVELS[lv-1].max:0;
  const top  = LEVELS[lv].max===9999? base+1000 : LEVELS[lv].max;
  return { curr: xp-base, range: top-base };
}

function renderHearts(id){
  const el=$(id); if(!el)return;
  el.innerHTML='';
  for(let i=0;i<MAX_HEARTS;i++){
    const h=document.createElement('span');
    h.className='heart'+(i>=G.hearts?' dead':'');
    h.textContent='❤️';
    el.appendChild(h);
  }
}

function updateCounts(){
  const cats=['all','html-struct','html-text','html-media','css-text','css-box','css-layout','css-visual'];
  cats.forEach(c=>{
    const n = c==='all'? QB.length : QB.filter(q=>q.c===c).length;
    const el=$('cnt-'+c); if(el) el.textContent=n;
  });
}

function updateTopBars(){
  const lv=G.level;
  const lvd=LEVELS[Math.min(lv,LEVELS.length-1)];
  const x=xpInLevel(G.xp,lv);
  const pct=Math.min(100,Math.round(x.curr/x.range*100));
  const acc = G.totalQ>0? Math.round(G.totalCorrect/G.totalQ*100)+'%':'—';

  const setTxt=(id,v)=>{const e=$(id);if(e)e.textContent=v;};
  setTxt('h-streak',G.streak); setTxt('h-xp',G.xp); setTxt('h-lv',lv+1);
  setTxt('h-lvname',lvd.name);
  setTxt('h-xpcurr',x.curr); setTxt('h-xpmax',x.range);
  setTxt('h-total-q',G.totalQ); setTxt('h-acc',acc); setTxt('h-streak2',G.streak);
  const bar=$('h-xpbar'); if(bar) bar.style.width=pct+'%';
  renderHearts('h-hearts');

  setTxt('q-streak',G.streak); setTxt('q-lv',lv+1); setTxt('q-xp',G.xp);
  renderHearts('q-hearts');

  setTxt('r-streak',G.streak); setTxt('r-xp',G.xp);
  renderHearts('r-hearts');
}

/* ═══════════════════════════════════════
   STREAK
═══════════════════════════════════════ */
function checkStreak(){
  const today=new Date().toDateString();
  if(G.lastDate===today) return;
  const yday=new Date(Date.now()-86400000).toDateString();
  G.streak = (G.lastDate===yday) ? G.streak+1 : 1;
  G.lastDate=today;
}

/* ═══════════════════════════════════════
   SESSION START
═══════════════════════════════════════ */
function startSession(){
  let pool = G.topic==='all' ? QB : QB.filter(q=>q.c===G.topic);
  if(G.mode!=='mixed') pool=pool.filter(q=>q.t===G.mode);
  if(!pool.length) pool=QB;
  const qs=shuffle(pool).slice(0,SESSION_LEN);
  G.session={ qs, idx:0, correct:0, xpGained:0, combo:0, maxCombo:0 };
  G.hearts=MAX_HEARTS;
  G.answered=false;
  G.pendingLevelUp=false;
  checkStreak();
  updateTopBars();
  showScreen('quizScreen');
  renderQ();
}

/* ═══════════════════════════════════════
   RENDER QUESTION
═══════════════════════════════════════ */
function renderQ(){
  G.answered=false;
  const s=G.session;
  const q=s.qs[s.idx];
  const total=s.qs.length;

  $('q-curr').textContent=s.idx+1;
  $('q-total').textContent=total;
  $('q-progfill').style.width=Math.round(s.idx/total*100)+'%';

  // type tag
  const tagMap={ flashcard:'🃏 플래시카드', blank:'⬜ 빈칸 채우기', typein:'⌨️ 직접 입력' };
  const clsMap={ flashcard:'qt-flash', blank:'qt-blank', typein:'qt-type' };
  const tt=$('qtTag');
  tt.textContent=tagMap[q.t]||q.t;
  tt.className='qtype-tag '+(clsMap[q.t]||'qt-flash');

  $('qQuestion').textContent=q.q;

  const codeEl=$('qCode');
  const optsEl=$('qOpts');
  const typeEl=$('qTypeWrap');
  const hintEl=$('qHint');
  const ti=$('typeInput');

  codeEl.style.display='none';
  optsEl.innerHTML='';
  optsEl.className='opts';
  typeEl.style.display='none';
  hintEl.textContent='';
  ti.value=''; ti.disabled=false;
  ti.className='type-input';

  if(q.t==='blank'){
    codeEl.style.display='block';
    codeEl.innerHTML=renderCode(q.code, q.blank);
  }

  if(q.t==='flashcard'||q.t==='blank'){
    const opts=shuffle(q.opts);
    if(opts.length<=2) optsEl.classList.add('single');
    opts.forEach(o=>{
      const b=document.createElement('button');
      b.className='opt';
      b.textContent=o;
      b.onclick=()=>selectOpt(b,o,q.ans);
      optsEl.appendChild(b);
    });
  } else {
    typeEl.style.display='flex';
    ti.focus();
  }

  if(q.hint) hintEl.textContent='💡 '+q.hint;

  $('feedback').className='feedback';
  $('nextBtn').className='next-btn';
  updateTopBars();
}

function renderCode(code, blank){
  let html=code
    .replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;')
    .replace(/___/g,`<span class="blank-slot">${blank}</span>`);
  // syntax highlight
  html=html
    .replace(/(&lt;\/?)([\w!]+)(&gt;)/g,'<span class="ck">$1$2$3</span>')
    .replace(/([\w-]+)(=)(&quot;)/g,'<span class="ca">$1</span>$2<span class="cv">$3')
    .replace(/(\d+)(px|em|rem|%|fr)/g,'<span class="cn">$1$2</span>');
  return html.replace(/\n/g,'<br>');
}

/* ═══════════════════════════════════════
   ANSWER HANDLING
═══════════════════════════════════════ */
function selectOpt(btn, sel, correct){
  if(G.answered) return;
  G.answered=true;
  document.querySelectorAll('.opt').forEach(b=>{
    b.disabled=true;
    if(b.textContent===correct) b.classList.add('correct');
  });
  const ok=(sel===correct);
  if(!ok) btn.classList.add('wrong');
  handleResult(ok, correct);
}

function submitType(){
  if(G.answered) return;
  const q=G.session.qs[G.session.idx];
  const val=$('typeInput').value.trim().toLowerCase();
  const accepts=(q.acc||[q.ans]).map(a=>a.toLowerCase());
  const ok=accepts.includes(val);
  G.answered=true;
  const ti=$('typeInput');
  ti.disabled=true;
  ti.classList.add(ok?'correct':'wrong');
  handleResult(ok,q.ans);
}

function handleResult(ok, correct){
  const s=G.session;
  const fb=$('feedback');

  if(ok){
    const xpBase=10;
    s.combo++;
    s.maxCombo=Math.max(s.maxCombo,s.combo);
    const bonus=s.combo>=3? Math.floor(s.combo/3)*5 : 0;
    const xp=xpBase+bonus;
    G.xp+=xp; G.totalCorrect++; G.totalQ++;
    s.correct++; s.xpGained+=xp;
    fb.className='feedback ok show';
    fb.innerHTML=`✓ 정답!  <strong>+${xp} XP</strong>${bonus?`  🔥 콤보 보너스 +${bonus}`:``}`;
    if(s.combo>=3) showCombo(s.combo);
    const newLv=calcLevel(G.xp);
    if(newLv>G.level){ G.level=newLv; G.pendingLevelUp=true; }
  } else {
    s.combo=0;
    G.hearts=Math.max(0,G.hearts-1);
    G.totalQ++;
    fb.className='feedback ng show';
    fb.innerHTML=`✗ 오답.  정답: <strong>${correct}</strong>`;
    if(G.hearts===0){ setTimeout(()=>endSession('hearts'),900); return; }
  }

  $('nextBtn').className='next-btn show';
  updateTopBars();
}

function showCombo(n){
  const el=$('comboFlash');
  el.textContent=`🔥 ${n} COMBO!`;
  el.classList.add('show');
  setTimeout(()=>el.classList.remove('show'),1500);
}

function nextQ(){
  if(G.pendingLevelUp){
    G.pendingLevelUp=false;
    showLevelUp();
    return;
  }
  G.session.idx++;
  if(G.session.idx>=G.session.qs.length){ endSession('done'); return; }
  renderQ();
}

/* ═══════════════════════════════════════
   END SESSION
═══════════════════════════════════════ */
function endSession(reason){
  const s=G.session;
  const pct=Math.round(s.correct/s.qs.length*100);
  showScreen('resultScreen');
  $('r-emoji').textContent=reason==='hearts'?'💔':pct>=90?'🏆':pct>=70?'🎉':pct>=50?'👍':'😅';
  $('r-title').textContent=reason==='hearts'?'목숨 소진!':'세션 완료!';
  $('r-sub').textContent=`${s.qs.length}문제 중 ${s.correct}개 정답 (${pct}%)`;
  $('r-correct').textContent=`${s.correct}/${s.qs.length}`;
  $('r-gained').textContent=`+${s.xpGained}`;
  $('r-combo').textContent=s.maxCombo;
  updateTopBars();
}

}
function closeModal(){

  G.session.idx++;
  if(G.session.idx>=G.session.qs.length){ endSession('done'); return; }
  renderQ();
}
